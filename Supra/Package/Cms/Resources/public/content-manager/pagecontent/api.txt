Supra.Manager.PageContent
	Methods:
		getIframe()
			Returns PageIframe object
			
		getContentContainer()
			Returns iframe contents object
		
		getContentBlocks()
			Returns all non-list content blocks
			
		getActiveContent()
			Returns active content block
		
		
		hasUnsavedChanges()
			Returns if there are unsaved changes
		
		
		startEditing()
			Change state to editing
		
		stopEditing()
			Change state to not editing
		
		isEditing()
			Returns if current state is editing


Supra.Manager.PageContent.PageIframe
	Attributes:
		overlayVisible
			Used to allow drag & drop from parent document
			
		html
			Iframe HTML content
		
		contentData
			Data about all contents 
		
		doc
			Iframe document object
		
		win
			Iframe window object
		
		
	Methods:
		addScript(src)
			Add script to the iframe content
			
		addStyleSheet(href)
			Add stylesheet to the iframe content


Supra.Manager.PageContent.IframeContents
	Attributes:
		iframe
			PageIframe instance
			
		contentData
			Data about all contents 
		
		doc
			Iframe document object
		
		win
			Iframe window object
		
		body
			Iframe body object
		
		activeContent
			Active content object
		
		highlight
			Highlight list nodes
			
	Methods:
		getBlockInsertData(data, callback, context)
			Loads data needed to insert new block and passes that data to callback
		
		removeChild(child)
			Remove children content
		
Supra.Manager.PageContent.Proto
	Events:
		editing-start
		editing-end
	
	Attributes:
		data
			Content data
			
		doc
			Iframe document object
		
		win
			Iframe window object
		
		body
			Iframe body object
		
		editing
			Current editing state
		
		parent
			Parent page content (Proto) or IframeContents object
		
		super
			IframeContents object
		
		editable
			Is content editable or not
		
		changed
			Has content changed since last save
		
		draggable
			Is content draggable (sort sorting) or not
			
		highlight
			If true content will be highlighted
			
	Methods:
		getType()
			Returns block type
			
		getId()
			Returns block instance ID
			
		getTitle()
			Returns block title
		
		getNodeId()
			Returns block container node ID attribute value
		
		getNode()
			Returns block container node (from iframe)
			
		getBlock()
			Returns block information
		
		getProperties()
			Returns all block properties
		
		isChildTypeAllowed(type)
			Returns if given type is allowed as child
		
		isClosed()
			Returns true if is closed (can't be editing, dragged or deleted)
			
		isParentClosed()
			Returns true if parent is closed (block can't be dragged, deleted and new blocks can't be added to parent)
		
		removeChild(child)
			Removes child block
			
		createChild(data, attrs)
			Creates child block
			
		syncOverlayPosition()
			Update overlay position to match block size
