Supra.SiteMap.Tree
	Attributes:
		mode
			Content mode, pages or templates 
	
		view
			Supra.SiteMap.TreeView instance
		
		data
			Supra.SiteMap.TreeData instance
		
		highlighted
			Supra.SiteMap.TreeNode instance or null
		
		loading
			Tree loading state
		
		locale
			Current locale
			
		visibleLayerDepth
			Show / hide layers above given depth
	
	Methods:
		item(id)
			If argument if a number then returns children node by index
			If argument is a string then returns descendant node by ID 
		
		children()
			Returns Y.Array with all children
		
		indexOf(id)
			Returns index of child node or -1 if not found
			If argument is a node then searches by node
			If argument is a string then searches by ID
		
		size()
			Returns children count
		
		insert(node, reference, where)
			Insert node 'before', 'after' or 'inside' reference node
			If first argument is data, then new node is created
		
		add(node, reference, where)
			Alias of insert
		
		remove(id)
			If argument is index, then removes node by index
			If argument is string, then removes node by ID
			If argument is node, then removes it
			
		removeAll()
			Removes all children nodes, recursive
		
		destroy()
			Destructor, removes and destroys all children nodes
	
	Events:
		page:select
			Fired when page is selected



Supra.SiteMap.TreeView
	Attributes:
		animating
			Animation is in progress
		
		
	Methods:
		center(id)
			Move element to be in the center of the screen
		
		scrollLeft()
			Scroll view left
		
		scrollRight()
			Scroll view right
		
		scrollX(x)
			Scroll view to X position
		
		scrollY(y)
			Scroll view to Y position
		
		resetCenter()
			Reset center position
		
		resetDropCache(clean)
			Reset drag and drop cache
			If clean argument is true then all drop targets are reseted
			
		checkOverflow()
			Update arrow visibility


Supra.SiteMap.TreeNode
	Attributes:
		depth
			Depth of the node relative to the tree
		
		index
			Index of the node in the parent children list
		
		parent
			Parent node or tree
		
		tree
			Tree
		
		children
			Y.Array of all children
		
		childrenRendered
			Children are rendered or not
		
		
		root
			This is a root page
			
		highlighted
			Tree node is highlighted (all other are darkened)
		
		expanded
			Expanded state
		
		selected
			Selected state
		
		selectable
			Page associated with node is selectable
		
		editable
			Page associated with node can be edited
		
		publishable
			Page associated with node can be published
		
		state
			Page state which is associated with node:
				draft, published, unpublished, global, temporary
		
		type
			Node type: page or application
		
		identifier
			ID assigned by tree, this is not page ID. To get page ID use data attribute
		
		label
			Label
		
		preview
			Preview URL
		
		fullPath
			Page full path
		
		data
			Page data
			
		draggable
			Node is draggable
		
		droppable
			Node is drop target
		
		dndMarker
			Node is marked as drop target, values are 'before', 'after', 'inside'
		
		dndLocked
			Node is locked, it can't be dragged or dropped
			even if 'draggable' and 'droppable' are true
		
		
	Methods:
		item(id)
			If argument if a number then returns children node by index
			If argument is a string then returns descendant node by ID
		
		next()
			Returns next sibling
			
		previous()
			Returns previous sibling
		
		children()
			Returns Y.Array with all rendered children
		
		indexOf(id)
			Returns index of child node or -1 if not found
			If argument is a node then searches by node
			If argument is a string then searches by ID
		
		size()
			Returns children count
		
		insert(node, where)
			Insert node before, after or inside current node
		
		add(node, where)
			Alias of insert
			
		remove(id)
			If argument is empty, then removes node itself
			If argument is index, then removes node by index
			If argument is string, then removes node by ID
			If argument is node, then removes it
			
		removeAll()
			Removes all children nodes, recursive
			
		destroy()
			Removes and destroys node and all children
		
		
		expand()
			Expand node
		
		collapse()
			Collapse node
		
		toggle()
			Toggle node
			
		
		edit()
			Open page editing window
		
		getWidget(id)
			Returns widget buttonOpen, buttonEdit by Id
		
		
		syncChildrenPosition()
			Update children container position, should be used if expand/collapse
			behaviour is overwritten
	
		updateFullPath()
			Update page full path, should be used after page parent changes
		
	Events:
		render:start
		render:complete
		renderChildren:start
		renderChildren:complete
		node:expand
		node:collapse



Supra.SiteMap.TreeData
	Attributes:
		requestURI
			Request URI
		
	Methods:
		
		item(id)
			Returns item data by ID
		
		children(id)
			Returns Y.Array with children data by parent ID
		
		all()
			Returns Y.Array of all data
		
		insert(data, reference, where)
			Insert data before/after/inside page data
			
		add(data, reference, where)
			Alias of insert
		
		remove(id)
			Removes item data by ID
		
		removeAll()
			Removes all data
			
		load()
			Load all data
			
		index(id)
			Returns index in parents children list
			
		
		
	Events:
		load:start
		load:complete
